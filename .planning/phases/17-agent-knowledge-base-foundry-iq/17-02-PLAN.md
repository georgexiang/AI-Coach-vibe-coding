### Task 2: API Client

Create `frontend/src/api/knowledge-base.ts`:

```typescript
export const knowledgeBaseApi = {
  listConnections: () => apiClient.get<SearchConnection[]>('/knowledge-base/connections'),
  listIndexes: () => apiClient.get<SearchIndex[]>('/knowledge-base/indexes'),
  getHcpConfigs: (hcpId: string) => apiClient.get<KnowledgeConfig[]>(`/knowledge-base/hcp/${hcpId}/configs`),
  addHcpConfig: (hcpId: string, data: KnowledgeConfigCreate) => apiClient.post(`/knowledge-base/hcp/${hcpId}/configs`, data),
  removeConfig: (configId: string) => apiClient.delete(`/knowledge-base/configs/${configId}`),
};
```
