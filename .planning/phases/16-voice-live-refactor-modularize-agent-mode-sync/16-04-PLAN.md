```typescript
import { describe, it, expect } from "vitest";
import { encodePcmToBase64 } from "../voice-utils";

describe("encodePcmToBase64", () => {
  it("encodes empty Float32Array to empty base64", () => {
    const result = encodePcmToBase64(new Float32Array(0));
    expect(result).toBe("");
  });

  it("encodes silence (zeros) correctly", () => {
    const silence = new Float32Array(4).fill(0);
    const result = encodePcmToBase64(silence);
    expect(typeof result).toBe("string");
    expect(result.length).toBeGreaterThan(0);
    // Decode and verify all zeros
    const decoded = atob(result);
    expect(decoded.length).toBe(8); // 4 samples * 2 bytes each
  });

  it("clips values above 1.0", () => {
    const loud = new Float32Array([1.5, 2.0, 100.0]);
    const result = encodePcmToBase64(loud);
    // Should clip to 1.0 -> 0x7FFF for each sample
    const clipped = encodePcmToBase64(new Float32Array([1.0, 1.0, 1.0]));
    expect(result).toBe(clipped);
  });

  it("clips values below -1.0", () => {
    const loud = new Float32Array([-1.5, -2.0, -100.0]);
    const result = encodePcmToBase64(loud);
    const clipped = encodePcmToBase64(new Float32Array([-1.0, -1.0, -1.0]));
    expect(result).toBe(clipped);
  });

  it("handles NaN and undefined values safely", () => {
    const bad = new Float32Array(3);
    bad[0] = NaN;
    expect(() => encodePcmToBase64(bad)).not.toThrow();
  });

  it("produces valid base64 output", () => {
    const audio = new Float32Array([0.5, -0.5, 0.0, 1.0, -1.0]);
    const result = encodePcmToBase64(audio);
    expect(() => atob(result)).not.toThrow();
  });
});
```

**3. Create `frontend/src/hooks/__tests__/use-voice-session-lifecycle.test.ts`:**

Test the reentrancy guard and unmount cancellation:

```typescript
import { describe, it, expect, vi, beforeEach } from "vitest";
import { renderHook, act } from "@testing-library/react";
import { useVoiceSessionLifecycle } from "../use-voice-session-lifecycle";

// Mock dependencies
const mockVoiceLive = {
  connect: vi.fn().mockResolvedValue({ avatarEnabled: false }),
  disconnect: vi.fn(),
  isConnected: false,
};
const mockAudioHandler = {
  initialize: vi.fn().mockResolvedValue(undefined),
  startRecording: vi.fn(),
  cleanup: vi.fn(),
};
const mockAvatarStream = {
  connect: vi.fn().mockResolvedValue(undefined),
  disconnect: vi.fn(),
};
const mockAudioPlayer = {
  stopAudio: vi.fn(),
};

describe("useVoiceSessionLifecycle", () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it("provides startSession and stopSession functions", () => {
    const { result } = renderHook(() =>
      useVoiceSessionLifecycle({
        voiceLive: mockVoiceLive as any,
        audioHandler: mockAudioHandler as any,
        avatarStream: mockAvatarStream as any,
        audioPlayer: mockAudioPlayer as any,
      })
    );

    expect(typeof result.current.startSession).toBe("function");
    expect(typeof result.current.stopSession).toBe("function");
    expect(result.current.isBusy).toBe(false);
  });

  it("prevents reentrancy (double startSession call)", async () => {
    mockVoiceLive.connect.mockImplementation(
      () => new Promise((resolve) => setTimeout(() => resolve({ avatarEnabled: false }), 1000))
    );

    const { result } = renderHook(() =>
      useVoiceSessionLifecycle({
        voiceLive: mockVoiceLive as any,
        audioHandler: mockAudioHandler as any,
        avatarStream: mockAvatarStream as any,
        audioPlayer: mockAudioPlayer as any,
      })
    );

    await act(async () => {
      result.current.startSession();
      result.current.startSession(); // Should be no-op due to reentrancy guard
    });

    expect(mockAudioHandler.initialize).toHaveBeenCalledTimes(1);
  });

  it("cancels in-flight connection on unmount", async () => {
    mockVoiceLive.connect.mockImplementation(
      () => new Promise((resolve) => setTimeout(() => resolve({ avatarEnabled: false }), 5000))
    );

    const { result, unmount } = renderHook(() =>
      useVoiceSessionLifecycle({
        voiceLive: mockVoiceLive as any,
        audioHandler: mockAudioHandler as any,
        avatarStream: mockAvatarStream as any,
        audioPlayer: mockAudioPlayer as any,
      })
    );

    await act(async () => {
      result.current.startSession();
    });

    unmount();

    expect(mockAudioHandler.cleanup).toHaveBeenCalled();
  });

  it("stopSession calls all cleanup functions", async () => {
    const { result } = renderHook(() =>
      useVoiceSessionLifecycle({
        voiceLive: mockVoiceLive as any,
        audioHandler: mockAudioHandler as any,
        avatarStream: mockAvatarStream as any,
        audioPlayer: mockAudioPlayer as any,
      })
    );

    await act(async () => {
      await result.current.stopSession();
    });

    expect(mockVoiceLive.disconnect).toHaveBeenCalled();
    expect(mockAvatarStream.disconnect).toHaveBeenCalled();
    expect(mockAudioHandler.cleanup).toHaveBeenCalled();
    expect(mockAudioPlayer.stopAudio).toHaveBeenCalled();
  });
});
```
