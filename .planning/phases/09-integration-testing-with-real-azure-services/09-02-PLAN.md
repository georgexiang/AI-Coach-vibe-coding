   }

   export async function updateAIFoundryConfig(config: AIFoundryConfigUpdate): Promise<AIFoundryConfig> {
     const { data } = await apiClient.put<AIFoundryConfig>("/azure-config/ai-foundry", config);
     return data;
   }

5. **TanStack Query hooks** (`frontend/src/hooks/use-azure-config.ts`): Add:
   ```typescript
   export function useAIFoundryConfig() {
     return useQuery({
       queryKey: ["ai-foundry-config"],
       queryFn: getAIFoundryConfig,
     });
   }

   export function useUpdateAIFoundry() {
     const queryClient = useQueryClient();
     return useMutation({
       mutationFn: updateAIFoundryConfig,
       onSuccess: () => {
         queryClient.invalidateQueries({ queryKey: ["ai-foundry-config"] });
         queryClient.invalidateQueries({ queryKey: ["service-configs"] });
       },
     });
   }
   ```
