```python
class CoachEventType(StrEnum):
    TEXT = "text"
    AUDIO = "audio"
    SCORE = "score"
    SUGGESTION = "suggestion"
    ERROR = "error"
    DONE = "done"

@dataclass
class CoachEvent:
    type: CoachEventType
    content: str
    metadata: dict | None = None

@dataclass
class CoachRequest:
    session_id: str
    message: str
    mode: str = "text"
    scenario_context: str = ""
    hcp_profile: dict | None = None
    scoring_criteria: dict | None = None

class BaseCoachingAdapter(ABC):
    name: str = ""
    @abstractmethod
    async def execute(self, request: CoachRequest) -> AsyncIterator[CoachEvent]: ...
    @abstractmethod
    async def is_available(self) -> bool: ...
```

   ```python
   class AzureOpenAIAdapter(BaseCoachingAdapter):
       name = "azure_openai"
   ```
   Constructor: `__init__(self, endpoint: str, api_key: str, deployment: str, api_version: str = "2024-06-01")`
   - Store as `self._endpoint`, `self._api_key`, `self._deployment`, `self._api_version`
   - Create client once in constructor: `from openai import AsyncAzureOpenAI` at top of `__init__`, store as `self._client`

   `async def execute(self, request: CoachRequest) -> AsyncIterator[CoachEvent]`:
   - Build messages array:
     - Start with `{"role": "system", "content": request.scenario_context}`
     - If `request.conversation_history` is provided, append all history messages
     - Append `{"role": "user", "content": request.message}`
   - Call `self._client.chat.completions.create(model=self._deployment, messages=messages, stream=True, temperature=0.7, max_tokens=1024)`
   - Iterate `async for chunk in stream`: if `chunk.choices and chunk.choices[0].delta.content`, yield `CoachEvent(type=CoachEventType.TEXT, content=chunk.choices[0].delta.content)`
   - After stream ends, yield `CoachEvent(type=CoachEventType.DONE, content="")`
   - Wrap entire body in try/except: on exception yield `CoachEvent(type=CoachEventType.ERROR, content=f"Azure OpenAI error: {str(e)}")` then `CoachEvent(type=CoachEventType.DONE, content="")`

   `async def is_available(self) -> bool`:
   - Return `bool(self._endpoint and self._api_key and self._deployment)`

   `async def get_version(self) -> str | None`:
   - Return `f"azure-openai-{self._api_version}"`

   Use conditional import pattern like Phase 06 Azure adapters: import `AsyncAzureOpenAI` inside `__init__` to avoid ImportError when openai is not installed (though it IS installed, this follows the project convention from stt/azure.py).
