1. Add ZIP export to `backend/app/services/skill_service.py`:

   **async def export_skill_package(db: AsyncSession, skill_id: str) -> bytes:**
   ```python
   import zipfile, io, json, yaml
   skill = await get_skill(db, skill_id)
   storage = get_storage()
   buffer = io.BytesIO()
   with zipfile.ZipFile(buffer, 'w', zipfile.ZIP_DEFLATED) as zf:
       # Reconstruct SKILL.md (YAML frontmatter + Markdown body)
       frontmatter = {
           "name": skill.name,
           "description": skill.description,
           "product": skill.product,
           "therapeutic_area": skill.therapeutic_area,
           "compatibility": skill.compatibility,
           "tags": skill.tags,
       }
       # Merge extra metadata
       try:
           extra = json.loads(skill.metadata_json) if skill.metadata_json else {}
           frontmatter.update(extra)
       except json.JSONDecodeError:
           pass
       skill_md = f"---\n{yaml.dump(frontmatter, allow_unicode=True, default_flow_style=False)}---\n\n{skill.content}"
       zf.writestr("SKILL.md", skill_md)
       
       # Resources (references/, scripts/, assets/)
       for resource in skill.resources:
           try:
               file_content = await storage.read(resource.storage_path)
               archive_path = f"{resource.resource_type}s/{resource.filename}"
               zf.writestr(archive_path, file_content)
           except Exception as e:
               logger.warning("Failed to include resource %s: %s", resource.filename, e)
   return buffer.getvalue()
   ```

2. Add ZIP import to `backend/app/services/skill_service.py`:

   **async def import_skill_package(db: AsyncSession, zip_data: bytes, user_id: str) -> Skill:**
   ```python
   import zipfile, io
   # Security: validate ZIP
   buffer = io.BytesIO(zip_data)
   if not zipfile.is_zipfile(buffer):
       raise bad_request("Invalid ZIP file")
   buffer.seek(0)
   
   with zipfile.ZipFile(buffer, 'r') as zf:
       # Security: check for path traversal (T-19-04)
       for name in zf.namelist():
           if '..' in name or name.startswith('/'):
               raise bad_request(f"Invalid path in ZIP: {name}")
       
       # Find SKILL.md
       skill_md_names = [n for n in zf.namelist() if n.endswith("SKILL.md")]
       if not skill_md_names:
           raise bad_request("No SKILL.md found in ZIP package")
       
       skill_md_content = zf.read(skill_md_names[0]).decode("utf-8")
       
       # Parse YAML frontmatter + Markdown body
       import yaml
       if skill_md_content.startswith("---"):
           parts = skill_md_content.split("---", 2)
           if len(parts) >= 3:
               frontmatter = yaml.safe_load(parts[1])
               body = parts[2].strip()
           else:
               frontmatter = {}
               body = skill_md_content
       else:
           frontmatter = {}
           body = skill_md_content
       
       # Create Skill from parsed data
       skill_data = SkillCreate(
           name=frontmatter.get("name", "Imported Skill"),
           description=frontmatter.get("description", ""),
           product=frontmatter.get("product", ""),
           therapeutic_area=frontmatter.get("therapeutic_area", ""),
           compatibility=frontmatter.get("compatibility", ""),
           tags=frontmatter.get("tags", ""),
           content=body,
       )
       skill = await create_skill(db, skill_data, user_id)
       
       # Import resource files
       storage = get_storage()
       for name in zf.namelist():
           if name.endswith("SKILL.md"):
               continue
           # Determine resource_type from path
           if name.startswith("references/"):
               resource_type = "reference"
           elif name.startswith("scripts/"):
               resource_type = "script"
           elif name.startswith("assets/"):
               resource_type = "asset"
           else:
               continue  # Skip unknown directories
           
           filename = name.split("/")[-1]
           if not filename:
               continue  # Skip directory entries
           
           file_content = zf.read(name)
           storage_path = f"skills/{skill.id}/{resource_type}s/{filename}"
           await storage.save(storage_path, file_content)
           
           resource = SkillResource(
               skill_id=skill.id,
               resource_type=resource_type,
               filename=filename,
               storage_path=storage_path,
               content_type=_guess_content_type(filename),
               file_size=len(file_content),
           )
           db.add(resource)
       
       db.flush()
       return skill
   ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| ZIP upload -> extract | Untrusted ZIP file from client -- path traversal risk |
| ZIP content -> DB | Imported SKILL.md YAML parsed -- injection risk |
| AI feedback -> LLM | User feedback text sent to Azure OpenAI |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-19-19 | Tampering | ZIP import path traversal | mitigate | Reject any ZIP entry with `..` in path or absolute paths. Only process entries under references/, scripts/, assets/ directories. |
| T-19-20 | Tampering | YAML frontmatter parsing | mitigate | Use yaml.safe_load() (not yaml.load()). Only extract known fields into Skill model. Unknown fields go to metadata_json. |
| T-19-21 | Denial of Service | ZIP bomb | mitigate | Max upload size 100MB. Max individual file 50MB. Max entries in ZIP: 500. |
| T-19-22 | Tampering | AI feedback injection | accept | Feedback text is sent as user content in LLM prompt. LLM output replaces SOP content, which admin reviews before publishing. Admin-only operation. |
</threat_model>
