3. Create `frontend/src/api/skills.ts`:
   ```typescript
   import apiClient from "./client";
   import type { Skill, SkillCreate, SkillUpdate, PaginatedSkills, StructureCheckResult, SkillEvaluationSummary } from "@/types/skill";

   export async function getSkills(params?: { page?: number; page_size?: number; status?: string; product?: string; search?: string }) {
     const { data } = await apiClient.get<PaginatedSkills>("/skills", { params });
     return data;
   }

   export async function getPublishedSkills(params?: { page?: number; page_size?: number; search?: string }) {
     const { data } = await apiClient.get<PaginatedSkills>("/skills/published", { params });
     return data;
   }

   export async function getSkill(id: string) {
     const { data } = await apiClient.get<Skill>(`/skills/${id}`);
     return data;
   }

   export async function createSkill(payload: SkillCreate) {
     const { data } = await apiClient.post<Skill>("/skills", payload);
     return data;
   }

   export async function updateSkill(id: string, payload: SkillUpdate) {
     const { data } = await apiClient.put<Skill>(`/skills/${id}`, payload);
     return data;
   }

   export async function deleteSkill(id: string) {
     await apiClient.delete(`/skills/${id}`);
   }

   export async function publishSkill(id: string) {
     const { data } = await apiClient.post<Skill>(`/skills/${id}/publish`);
     return data;
   }

   export async function archiveSkill(id: string) {
     const { data } = await apiClient.post<Skill>(`/skills/${id}/archive`);
     return data;
   }

   export async function restoreSkill(id: string) {
     const { data } = await apiClient.post<Skill>(`/skills/${id}/restore`);
     return data;
   }

   export async function startConversion(id: string) {
     const { data } = await apiClient.post(`/skills/${id}/convert`);
     return data;
   }

   export async function retryConversion(id: string) {
     const { data } = await apiClient.post(`/skills/${id}/retry-conversion`);
     return data;
   }

   export async function getConversionStatus(id: string) {
     const { data } = await apiClient.get<{ conversion_status: string; conversion_error: string }>(`/skills/${id}/conversion-status`);
     return data;
   }

   export async function checkStructure(id: string) {
     const { data } = await apiClient.post<StructureCheckResult>(`/skills/${id}/check-structure`);
     return data;
   }

   export async function evaluateQuality(id: string) {
     const { data } = await apiClient.post(`/skills/${id}/evaluate-quality`);
     return data;
   }

   export async function getEvaluation(id: string) {
     const { data } = await apiClient.get<SkillEvaluationSummary>(`/skills/${id}/evaluation`);
     return data;
   }

   export async function uploadResources(id: string, files: File[], resourceType: string = "reference") {
     const formData = new FormData();
     files.forEach(f => formData.append("files", f));
     const { data } = await apiClient.post(`/skills/${id}/resources?resource_type=${resourceType}`, formData, {
       headers: { "Content-Type": "multipart/form-data" },
     });
     return data;
   }

   export async function uploadAndConvert(id: string, files: File[]) {
     const formData = new FormData();
     files.forEach(f => formData.append("files", f));
     const { data } = await apiClient.post(`/skills/${id}/upload-and-convert`, formData, {
       headers: { "Content-Type": "multipart/form-data" },
     });
     return data;
   }

   export async function exportSkillZip(id: string) {
     const { data } = await apiClient.get(`/skills/${id}/export`, { responseType: "blob" });
     return data;
   }
   ```

4. Create `frontend/src/hooks/use-skills.ts`:
   ```typescript
   import { useMutation, useQuery, useQueryClient } from "@tanstack/react-query";
   import * as skillsApi from "@/api/skills";
   import type { SkillCreate, SkillUpdate } from "@/types/skill";

   const SKILLS_KEY = "skills";

   export function useSkills(params?: { page?: number; page_size?: number; status?: string; product?: string; search?: string }) {
     return useQuery({ queryKey: [SKILLS_KEY, params], queryFn: () => skillsApi.getSkills(params) });
   }

   export function usePublishedSkills(params?: { page?: number; page_size?: number; search?: string }) {
     return useQuery({ queryKey: [SKILLS_KEY, "published", params], queryFn: () => skillsApi.getPublishedSkills(params) });
   }

   export function useSkill(id: string | undefined) {
     return useQuery({ queryKey: [SKILLS_KEY, id], queryFn: () => skillsApi.getSkill(id!), enabled: !!id });
   }

   export function useCreateSkill() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (data: SkillCreate) => skillsApi.createSkill(data), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useUpdateSkill() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: ({ id, data }: { id: string; data: SkillUpdate }) => skillsApi.updateSkill(id, data), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useDeleteSkill() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.deleteSkill(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function usePublishSkill() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.publishSkill(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useArchiveSkill() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.archiveSkill(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useRestoreSkill() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.restoreSkill(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useConversionStatus(id: string | undefined, enabled: boolean = false) {
     return useQuery({ queryKey: [SKILLS_KEY, id, "conversion"], queryFn: () => skillsApi.getConversionStatus(id!), enabled: !!id && enabled, refetchInterval: 3000 });
   }

   export function useStartConversion() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.startConversion(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useRetryConversion() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.retryConversion(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useCheckStructure() {
     const qc = useQueryClient();
     return useMutation({ mutationFn: (id: string) => skillsApi.checkStructure(id), onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }) });
   }

   export function useEvaluateQuality() {
     return useMutation({ mutationFn: (id: string) => skillsApi.evaluateQuality(id) });
   }

   export function useSkillEvaluation(id: string | undefined) {
     return useQuery({ queryKey: [SKILLS_KEY, id, "evaluation"], queryFn: () => skillsApi.getEvaluation(id!), enabled: !!id });
   }

   export function useUploadResources() {
     const qc = useQueryClient();
     return useMutation({
       mutationFn: ({ id, files, resourceType }: { id: string; files: File[]; resourceType?: string }) =>
         skillsApi.uploadResources(id, files, resourceType),
       onSuccess: () => qc.invalidateQueries({ queryKey: [SKILLS_KEY] }),
     });
   }
   ```
