```python
import pytest
from httpx import AsyncClient

# Test 1: Preview instructions with auto-generated text (no override)
@pytest.mark.asyncio
async def test_preview_instructions_auto_generated(admin_client: AsyncClient):
    """POST /preview-instructions returns auto-generated instructions when no override."""
    response = await admin_client.post(
        "/api/v1/hcp-profiles/preview-instructions",
        json={
            "name": "Dr. Test",
            "specialty": "Oncology",
            "hospital": "Test Hospital",
            "personality_type": "friendly",
            "emotional_state": 30,
            "communication_style": 70,
            "expertise_areas": ["Breast Cancer", "Immunotherapy"],
            "objections": ["Cost concerns"],
            "agent_instructions_override": "",
        },
    )
    assert response.status_code == 200
    data = response.json()
    assert "instructions" in data
    assert "is_override" in data
    assert data["is_override"] is False
    assert "Dr. Test" in data["instructions"]
    assert "Oncology" in data["instructions"]

# Test 2: Preview instructions with override text
@pytest.mark.asyncio
async def test_preview_instructions_with_override(admin_client: AsyncClient):
    """POST /preview-instructions returns override text when provided."""
    override_text = "You are a custom agent with special instructions."
    response = await admin_client.post(
        "/api/v1/hcp-profiles/preview-instructions",
        json={
            "name": "Dr. Override",
            "specialty": "Hematology",
            "agent_instructions_override": override_text,
        },
    )
    assert response.status_code == 200
    data = response.json()
    assert data["is_override"] is True
    assert data["instructions"] == override_text

# Test 3: Preview instructions requires admin auth
@pytest.mark.asyncio
async def test_preview_instructions_unauthorized(client: AsyncClient):
    """POST /preview-instructions rejects unauthenticated request."""
    response = await client.post(
        "/api/v1/hcp-profiles/preview-instructions",
        json={"name": "Test"},
    )
    assert response.status_code in (401, 403)

# Test 4: Preview instructions with empty body uses defaults
@pytest.mark.asyncio
async def test_preview_instructions_empty_body(admin_client: AsyncClient):
    """POST /preview-instructions works with all-default values."""
    response = await admin_client.post(
        "/api/v1/hcp-profiles/preview-instructions",
        json={},
    )
    assert response.status_code == 200
    data = response.json()
    assert data["is_override"] is False
    assert isinstance(data["instructions"], str)
    assert len(data["instructions"]) > 0
```
