    Keep backward compatibility with existing `AdapterRegistry` but extend to support categories:
    ```python
    class ServiceRegistry:
        """Singleton registry managing multiple adapter categories."""
        _instance = None
        _categories: dict[str, dict[str, object]] = {}  # category -> {name: adapter}

        def __new__(cls):
            if cls._instance is None:
                cls._instance = super().__new__(cls)
                cls._categories = {}
            return cls._instance

        def register(self, category: str, adapter: object) -> None:
            if category not in self._categories:
                self._categories[category] = {}
            self._categories[category][adapter.name] = adapter

        def get(self, category: str, name: str) -> object | None:
            return self._categories.get(category, {}).get(name)

        def list_category(self, category: str) -> list[str]:
            return list(self._categories.get(category, {}).keys())

        async def discover_category(self, category: str) -> list[str]:
            available = []
            for name, adapter in self._categories.get(category, {}).items():
                if await adapter.is_available():
                    available.append(name)
            return available

        def list_all_categories(self) -> dict[str, list[str]]:
            return {cat: list(adapters.keys()) for cat, adapters in self._categories.items()}

    # Backward compat alias
    AdapterRegistry = ServiceRegistry
    registry = ServiceRegistry()
    ```

    **Create `backend/app/api/config.py`** with config/features endpoint:
    ```python
    from fastapi import APIRouter, Depends
    from pydantic import BaseModel
    from app.config import get_settings
    from app.dependencies import get_current_user
    from app.services.agents.registry import registry

    router = APIRouter(prefix="/config", tags=["config"])

    class FeatureFlags(BaseModel):
        avatar_enabled: bool
        voice_enabled: bool
        realtime_voice_enabled: bool
        conference_enabled: bool
        default_voice_mode: str
        region: str

    class ConfigResponse(BaseModel):
        features: FeatureFlags
        available_adapters: dict[str, list[str]]

    @router.get("/features", response_model=ConfigResponse)
    async def get_features(user=Depends(get_current_user)):
        settings = get_settings()
        adapters = registry.list_all_categories()
        return ConfigResponse(
            features=FeatureFlags(
                avatar_enabled=settings.feature_avatar_enabled,
                voice_enabled=settings.feature_voice_enabled,
                realtime_voice_enabled=settings.feature_realtime_voice_enabled,
                conference_enabled=settings.feature_conference_enabled,
                default_voice_mode=settings.default_voice_mode,
                region=settings.region,
            ),
            available_adapters=adapters,
        )
    ```
