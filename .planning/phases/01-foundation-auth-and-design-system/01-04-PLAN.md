    **Create `frontend/src/stores/auth-store.ts`:**
    Lightweight store using a module-level state object with getters/setters and a subscriber pattern for React re-renders. OR use a simple React context approach. The store must:
    - Read `access_token` from `localStorage` on init (session persistence per AUTH-02)
    - Expose `token: string | null`, `user: User | null`, `isAuthenticated: boolean`
    - `setAuth(token: string, user: User)`: stores token in localStorage, updates state
    - `clearAuth()`: removes token from localStorage, clears state
    - Export `useAuthStore` hook for React components
    - Use `useSyncExternalStore` or `useState` with context for reactivity

    Implementation approach: Create a simple store using `useSyncExternalStore`:
    ```typescript
    let authState: AuthState = {
      token: localStorage.getItem("access_token"),
      user: null,
    };
    const listeners = new Set<() => void>();
    function subscribe(listener: () => void) { listeners.add(listener); return () => listeners.delete(listener); }
    function getSnapshot() { return authState; }
    function setAuth(token: string, user: User) {
      localStorage.setItem("access_token", token);
      authState = { token, user };
      listeners.forEach(l => l());
    }
    function clearAuth() {
      localStorage.removeItem("access_token");
      authState = { token: null, user: null };
      listeners.forEach(l => l());
    }
    export function useAuthStore() {
      const state = useSyncExternalStore(subscribe, getSnapshot);
      return { ...state, isAuthenticated: !!state.token, setAuth, clearAuth };
    }
    ```

    **Create `frontend/src/hooks/use-auth.ts`:**
    TanStack Query hooks for auth operations (per CLAUDE.md: no inline useQuery):
    - `useLogin()`: mutation that POSTs to `/auth/login`, on success calls `setAuth(token, user)` after fetching `/auth/me`
    - `useMe()`: query that GETs `/auth/me`, enabled only when token exists, on success updates user in store, on 401 clears auth
    - `useLogout()`: function that calls `clearAuth()` and navigates to `/login`
    ```typescript
    import { useMutation, useQuery } from "@tanstack/react-query";
    import apiClient from "@/api/client";
    import { useAuthStore } from "@/stores/auth-store";
    import type { LoginRequest, TokenResponse, User } from "@/types/auth";

    export function useLogin() {
      const { setAuth } = useAuthStore();
      return useMutation({
        mutationFn: async (data: LoginRequest) => {
          const res = await apiClient.post<TokenResponse>("/auth/login", data);
          localStorage.setItem("access_token", res.data.access_token);
          const meRes = await apiClient.get<User>("/auth/me");
          setAuth(res.data.access_token, meRes.data);
          return meRes.data;
        },
      });
    }

    export function useMe() {
      const { token, setAuth, clearAuth } = useAuthStore();
      return useQuery({
        queryKey: ["auth", "me"],
        queryFn: async () => {
          const res = await apiClient.get<User>("/auth/me");
          if (token) setAuth(token, res.data);
          return res.data;
        },
        enabled: !!token,
        retry: false,
      });
    }
    ```

    ```typescript
    import { Navigate, Outlet } from "react-router-dom";
    import { useAuthStore } from "@/stores/auth-store";
    import { useMe } from "@/hooks/use-auth";

    export function ProtectedRoute() {
      const { token } = useAuthStore();
      const { isLoading } = useMe();
      if (!token) return <Navigate to="/login" replace />;
      if (isLoading) return <div className="flex h-screen items-center justify-center">Loading...</div>;
      return <Outlet />;
    }

    export function AdminRoute() {
      const { user } = useAuthStore();
      if (user?.role !== "admin") return <Navigate to="/user/dashboard" replace />;
      return <Outlet />;
    }

    export function GuestRoute() {
      const { token } = useAuthStore();
      if (token) return <Navigate to="/user/dashboard" replace />;
      return <Outlet />;
    }
    ```
